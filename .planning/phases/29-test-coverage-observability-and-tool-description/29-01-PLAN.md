function createLogCapture(): { logs: LogEntry[]; stream: Writable } {
  const logs: LogEntry[] = [];
  const stream = new Writable({
    write(chunk, _encoding, callback) {
      try { logs.push(JSON.parse(chunk.toString().trim())); } catch {}
      callback();
    },
  });
  return { logs, stream };
}
